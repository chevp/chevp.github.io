1 Figma Make Export

Figma Make exportiert Design-Prototypen als statisches HTML/CSS. Der MCP-Server analysiert diese Struktur und extrahiert Komponenten-Hierarchien, Styles und Asset-Referenzen.

Erwartete Export-Struktur:

figma-export/
├── index.html          # Main entry point
├── styles/
│   ├── global.css       # Design tokens, variables
│   └── components.css   # Component styles
├── assets/
│   ├── images/          # Exported images
│   ├── icons/           # SVG icons
│   └── fonts/           # Custom fonts
└── pages/
    ├── dashboard.html   # Individual pages
    ├── login.html
    └── settings.html

Figma Component Naming Convention:

Figma-Komponenten sollten mit speziellen Prefixen benannt werden, um die automatische Erkennung zu ermoeglichen:

@form/

Formular-Elemente mit Validierung

@data/

Daten-gebundene Komponenten

@action/

Buttons mit API-Calls

@nav/

Navigation/Routing

description Beispiel: use-cases/user-authentication.md 

# User Authentication

## Overview
Ermoeglicht Benutzern die Anmeldung mit E-Mail und Passwort.

## UI Components
- @form/LoginForm: Hauptformular fuer Login
- @action/SubmitButton: Absende-Button
- @data/ErrorMessage: Fehleranzeige

## API Binding
```yaml
endpoint: POST /api/auth/login
request:
  email: string (required, email format)
  password: string (required, min 8 chars)
response:
  success:
    token: string (JWT)
    user: UserProfile
  error:
    code: string
    message: string
```

## Validation Rules
| Field    | Rule                | Error Message              |
|----------|---------------------|----------------------------|
| email    | required, email     | "Gueltige E-Mail eingeben" |
| password | required, min:8     | "Mind. 8 Zeichen"          |

## State Transitions
1. IDLE -> LOADING (on submit)
2. LOADING -> SUCCESS (on 200)
3. LOADING -> ERROR (on 4xx/5xx)
4. SUCCESS -> redirect to /dashboard

## Error Handling
- 401: "Falsche Anmeldedaten"
- 429: "Zu viele Versuche"
- 500: "Serverfehler, bitte spaeter versuchen"

code Beispiel: Generierte Login-Komponente 

import { Component, inject } from '@angular/core';
import { FormBuilder, ReactiveFormsModule, Validators } from '@angular/forms';
import { Router } from '@angular/router';
import { AuthService } from '@core/services/auth.service';

@Component({
  selector: 'app-login',
  standalone: true,
  imports: [ReactiveFormsModule],
  templateUrl: './login.component.html',  // From Figma
  styleUrl: './login.component.scss'       // From Figma
})
export class LoginComponent {
  private fb = inject(FormBuilder);
  private authService = inject(AuthService);
  private router = inject(Router);

  // Form mit Validierung aus use-case spec
  loginForm = this.fb.group({
    email: ['', [Validators.required, Validators.email]],
    password: ['', [Validators.required, Validators.minLength(8)]]
  });

  // State aus workflow definition
  state: 'idle' | 'loading' | 'success' | 'error' = 'idle';
  errorMessage = '';

  // Error messages aus validation spec
  readonly errorMessages = {
    email: 'Gueltige E-Mail eingeben',
    password: 'Mind. 8 Zeichen'
  };

  async onSubmit() {
    if (this.loginForm.invalid) return;

    this.state = 'loading';

    try {
      // API call aus endpoint spec
      const response = await this.authService.login(
        this.loginForm.value as LoginRequest
      );

      this.state = 'success';
      this.router.navigate(['/dashboard']);  // From state transition

    } catch (error) {
      this.state = 'error';
      // Error handling aus spec
      this.errorMessage = this.getErrorMessage(error.status);
    }
  }

  private getErrorMessage(status: number): string {
    const messages: Record<number, string> = {
      401: 'Falsche Anmeldedaten',
      429: 'Zu viele Versuche',
      500: 'Serverfehler, bitte spaeter versuchen'
    };
    return messages[status] ?? 'Unbekannter Fehler';
  }
}

api REST API Binding

Die REST API wird ueber OpenAPI/Swagger spezifiziert. Der MCP-Server generiert daraus typisierte Angular Services mit allen Endpoints, Models und Error Handling.

OpenAPI Beispiel (specs/api/openapi.yaml):

openapi: '3.0.3'
info:
  title: 'E-Commerce API'
  version: '1.0.0'

servers:
  - url: 'https://api.example.com/v1'

paths:
  /auth/login:
    post:
      operationId: login
      tags: [Authentication]
      summary: 'User login with email and password'
      requestBody:
        required: true
        content:
          application/json:
            schema:
              $ref: '#/components/schemas/LoginRequest'
      responses:
        '200':
          description: 'Successful login'
          content:
            application/json:
              schema:
                $ref: '#/components/schemas/AuthResponse'
        '401':
          description: 'Invalid credentials'
        '429':
          description: 'Too many requests'

  /products:
    get:
      operationId: listProducts
      tags: [Products]
      parameters:
        - name: category
          in: query
          schema:
            type: string
        - name: page
          in: query
          schema:
            type: integer
            default: 1

components:
  schemas:
    LoginRequest:
      type: object
      required: [email, password]
      properties:
        email:
          type: string
          format: email
        password:
          type: string
          minLength: 8

    AuthResponse:
      type: object
      properties:
        token:
          type: string
        user:
          $ref: '#/components/schemas/UserProfile'

Generierter Angular Service:

// Generated from OpenAPI spec
import { Injectable, inject } from '@angular/core';
import { HttpClient, HttpParams } from '@angular/common/http';
import { Observable } from 'rxjs';
import { environment } from '@env/environment';

// Generated types
export interface LoginRequest {
  email: string;
  password: string;
}

export interface AuthResponse {
  token: string;
  user: UserProfile;
}

export interface UserProfile {
  id: string;
  email: string;
  name: string;
}

@Injectable({ providedIn: 'root' })
export class AuthService {
  private http = inject(HttpClient);
  private baseUrl = environment.apiUrl;

  /**
   * User login with email and password
   * POST /auth/login
   */
  login(request: LoginRequest): Observable {
    return this.http.post(
      `${this.baseUrl}/auth/login`,
      request
    );
  }

  /**
   * Promise-based login for async/await usage
   */
  async loginAsync(request: LoginRequest): Promise {
    return firstValueFrom(this.login(request));
  }
}
